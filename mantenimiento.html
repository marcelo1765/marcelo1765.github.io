📝 Ejemplo de Código - Análisis de Ventas

CREATE PROCEDURE sp_ReporteVentasMensuales
    @Year INT,
    @Month INT = NULL
AS
BEGIN
    WITH VentasCTE AS (
        SELECT
            YEAR(fecha_venta) AS año,
            MONTH(fecha_venta) AS mes,
            SUM(monto_total) AS total_ventas,
            COUNT(DISTINCT id_cliente) AS clientes_activos
        FROM ventas
        WHERE YEAR(fecha_venta) = @Year
            AND (@Month IS NULL OR MONTH(fecha_venta) = @Month)
        GROUP BY YEAR(fecha_venta), MONTH(fecha_venta)
    )
    SELECT *,
        LAG(total_ventas) OVER (ORDER BY mes) AS ventas_mes_anterior,
        CAST(
            (LEAD(total_ventas) OVER (ORDER BY mes) - total_ventas) 
            AS DECIMAL(10,2)
        ) AS crecimiento_proyectado
    FROM VentasCTE
    ORDER BY mes;
END;
